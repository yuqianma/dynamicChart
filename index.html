Usage:

var chart = Chart('chart-container', [

{x: 1, y: 2, z: 3},

{x: 4, y: 5, z: 6}

]);

// chart-container is a target div

// take in an array contains points

chart.register(['peek', 'select', 'wheel']);

// register or deregister event(s)

chart.addPoint({x: 7, y: 8, z: 9});

// add a point
